### Replicate Function

```sh
> # number replicate
> x <- rep(1, times=4) # 1 1 1 1
> x
[1] 1 1 1 1
> # character replicate
> c <- rep('z', times=5)
> c
[1] "z" "z" "z" "z" "z"
> # sequence replicate
> x = 1 : 3
> y = rep(x, times=3)
> y
[1] 1 2 3 1 2 3 1 2 3
> # each arguments
> x = 1 : 3
> y = rep(x, each=3) # each element assign 3 times
> y
[1] 1 1 1 2 2 2 3 3 3
```

### `if-else` (same as C/Java/C++)

```r
x <- 0

if (x > 0) {
    print("x is positive")
} else if (x == 0) {
    print("x is zero")
} else {
    print("number is negative")
}
print("End of the program")
```

### `ifelse()` function

```r
x <- 10

# ifelse(condition, 'true', 'false')
# check number is positive or negative
ifelse(x >= 0, 'Number is Positive', 'Number is Negative')

# check odd and even number
ifelse(x%%2==0, 'Number is even', 'Number is odd')
```

### `repeat` loops

```r
i <- 1

# print 1 ... 5
# in repeat loop we need break statement 
repeat {
    print(i)
    if (i > 5) {
        break
    }
    i <- i + 1
}
```

### Function Returning Multiple Value

```r
# function return multiple value
myeval = function(x, y) {
    add = x + y
    mul = x * y
    result = list('sum'=add, 'mul'=mul)
    return(result)
}

print(myeval(10,2))
```

### inline function

```r
mysum = function(x, y) x + y
myexp = function(x, y) x ^ y

print(mysum(10, 20))
print(myexp(2,3))
```
